rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // Utility function: check if user is admin
    function isAdmin(uid) {
      return get(/databases/$(database)/documents/users/$(uid)).data.role == 'admin';
    }

    // Users collection
    match /users/{userId} {
      // anyone signed‑in can read their own doc or, if they're an admin,
      // read any user.
      allow read: if request.auth != null &&
                  (request.auth.uid == userId || isAdmin(request.auth.uid));

      // creation: only the authenticated user may create their document
      allow create: if request.auth != null &&
                    request.auth.uid == userId;

      // updates: owner may modify their document but may not change the role
      // field; an admin may modify anything (including role changes).
      allow update: if request.auth != null && (
           (request.auth.uid == userId &&
              request.resource.data.role == resource.data.role
           )
           || isAdmin(request.auth.uid)
      );

      // deletions are disallowed by clients
      allow delete: if false;
    }

    // block everything else by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
